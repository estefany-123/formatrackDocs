import { Tabs, TabItem } from '@astrojs/starlight/components';

Este endpoint permite actualizar la información de un área específica mediante su ID.

Method: `PUT`

<Tabs>
  <TabItem label="Endpoint">
  ```
  http://localhost:3000/Areas/:id_Area
  ```
  </TabItem>
</Tabs>

<h3>Encabezados de Solicitud</h3>

| Encabezado | Valor | Descripcion
|----------|-------| -------
| Authorization |  Bearer `token` | Token con autorizacion para acceder a las areas


### Body

```json
{
  "nombre": "string",
  "persona_encargada": "string",
  "estado": "boolean",
  "fecha_creacion": "date",
  "fecha_actualizacion": "date",
  "fk_sede": "integer"
}
```

<h3>Ejemplos de respuesta</h3>

<Tabs>
  <TabItem label="Status 200">
  ```json
  {message:"Se actualizo el area correctamente"}
  ```
  </TabItem>

  <TabItem label="Status 400">
  ```json
  {message:"No se actualizo el area"}
  ```
  </TabItem>
</Tabs>
