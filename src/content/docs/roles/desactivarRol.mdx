import { Tabs, TabItem } from '@astrojs/starlight/components';

Este endpoint permite cambiar el estado de un rol (activo/inactivo) utilizando su ID.

Method : `PUT`

<Tabs>
  <TabItem label="Endpoint">
  ```
  http://localhost:3000/api/rol/desactivar/:id_rol
  ```
  </TabItem>
</Tabs>


<h3>Encabezados de Solicitud</h3>

| Encabezado | Valor | Descripcion
|----------|-------| -------
| Authorization |  Bearer `token` | Token con autorizacion para acceder a los roles



<h3>Ejemplos de respuesta</h3>

<Tabs>
  <TabItem label="Status 201">
  ```json
  {message:"el rol se ha desactivado correctamente"}
  ```
  </TabItem>

  <TabItem label="Status 400">
  ```json
  {message:"No fue posible desactivar el rol"}
  ```
  </TabItem>
</Tabs>
