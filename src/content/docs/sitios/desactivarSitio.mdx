import { Tabs, TabItem } from '@astrojs/starlight/components';

Este endpoint permite cambiar el estado de un sitio (activo/inactivo) utilizando su ID.

Method : `PUT`

<Tabs>
  <TabItem label="Endpoint">
  ```
  http://localhost:3000/Sitio/estado/:id_sitio
  ```
  </TabItem>
</Tabs>


<h3>Encabezados de Solicitud</h3>

| Encabezado | Valor | Descripcion
|----------|-------| -------
| Authorization |  Bearer `token` | Token con autorizacion para acceder a los sitios



<h3>Ejemplos de respuesta</h3>

<Tabs>
  <TabItem label="Status 200">
  ```json
  {message:"Se desactivo el sitio correctamente"}
  ```
  </TabItem>

  <TabItem label="Status 400">
  ```json
  {message:"No se desactivo el sitio"}
  ```
  </TabItem>
</Tabs>
