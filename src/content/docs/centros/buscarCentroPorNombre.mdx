import { Tabs, TabItem } from '@astrojs/starlight/components';


Este endpoint permite buscar un centro específico por su nombre.

Methodo:`GET`

<h3>Endpoint</h3>

<Tabs>
  <TabItem label="Endpoint">
  ```
  http://localhost:3000/centros/:nombre
  ```
  </TabItem>
</Tabs>

<h3>Encabezados de Solicitud</h3>

| Encabezado | Valor | Descripcion
|----------|-------| -------
| Authorization |  Bearer `token` | Token con autorizacion para acceder a las areas

<h3>Ejemplo de respuesta</h3>

<Tabs>
  <TabItem label="Status 200">
  ```json
{
    "id_centro": 1,
    "nombre": "CGDSS1",
    "estado": true,
    "fecha_creacion": "2026-02-08T05:00:00.000Z",
    "fecha_actualizacion": "2024-05-08T05:00:00.000Z",
    "fk_municipio": 1
  }
```
  </TabItem>

  <TabItem label="Status 404">
  ```json
  {message:"No se encontró ningún centro con el nombre ingresado"}
  ```
  </TabItem>

</Tabs>
