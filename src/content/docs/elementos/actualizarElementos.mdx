import { Tabs, TabItem } from '@astrojs/starlight/components';

Este endpoint permite actualizar la información de un elemento específico mediante su ID.

Method: `PUT`


<Tabs>
  <TabItem label="Endpoint">
  ```
  http://localhost:3000/api/elemento/update/:id_elemento
  ```
  </TabItem>
</Tabs>

<h3>Encabezados de Solicitud</h3>

| Encabezado | Valor | Descripcion
|----------|-------| -------
| Authorization |  Bearer `token` | Token con autorizacion para acceder a los elementos


### Body

```json
{
  "nombre": "string",
  "descripcion": "string",
  "valor": "integer",
  "consumible": "boolean",
  "no_consumible": "boolean",
  "estado": "boolean",
  "imagen_elemento": "string",
  "created_at": "date",
  "updated_at": "date",
  "fk_unidad_medida": "integer",
  "fk_categoria": "integer",
  "fk_caracteristica": "integer"
}
```


<h3>Ejemplos de respuesta</h3>

<Tabs>
  <TabItem label="Status 201">
  ```json
  {message:"Se actualizo el elemento correctamente"}
  ```
  </TabItem>

  <TabItem label="Status 401">
  ```json
  {message:"No se actualizo el elemento"}
  ```
  </TabItem>
</Tabs>
